    <script>

        let images = [

            "obrazy/barirlandzki.jpeg",

            "obrazy/czerwonabluza.jpeg",

            "obrazy/fajnaczapa.jpeg",

            "obrazy/stretching.jpeg"

        ];

        let currentImageIndex = 0;


        // Funkcja do wyświetlania zakładek

        function showTab(tab) {

            let tabs = document.querySelectorAll('.tab-content');

            tabs.forEach(tab => tab.classList.remove('active'));

            document.getElementById(tab).classList.add('active');

        }


        // Funkcja do zmiany zdjęcia w galerii

        function changeImage(direction) {

            if (direction === 'left') {

                currentImageIndex = (currentImageIndex === 0) ? images.length - 1 : currentImageIndex - 1;

            } else if (direction === 'right') {

                currentImageIndex = (currentImageIndex === images.length - 1) ? 0 : currentImageIndex + 1;

            }

            document.getElementById('currentImage').src = images[currentImageIndex];

        }


        // Funkcja do zapisania się na newsletter

        function submitEmail() {

            let email = document.getElementById('email').value;

            if (email) {

                alert(`Dziękujemy za zapisanie się do newslettera! Twój e-mail: ${email}`);

                document.getElementById('newsletterPopup').style.display = 'none';

            } else {

                alert('Proszę wprowadzić poprawny adres e-mail.');

            }

        }


        // Otwórz popup z newsletterem po 5 sekundach

        setTimeout(() => {

            document.getElementById('newsletterPopup').style.display = 'flex';

        }, 5000);
